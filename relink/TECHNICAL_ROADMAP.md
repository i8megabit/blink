import asyncio
import aiohttp
from bs4 import BeautifulSoup
from urllib.parse import urljoin, urlparse
from app import db
from app.models import Domain, Analysis
from app.utils.ollama_client import OllamaClient
from app.utils.cache import cache

class SEOAnalyzer:
    """–ù–∞—Ç–∏–≤–Ω—ã–π SEO –∞–Ω–∞–ª–∏–∑–∞—Ç–æ—Ä —Å –∞—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–æ–π –æ–±—Ä–∞–±–æ—Ç–∫–æ–π"""

        self.session = None
    
    async def __aenter__(self):
        self.session = aiohttp.ClientSession()
        return self
    
    async def __aexit__(self, exc_type, exc_val, exc_tb):
        if self.session:
            await self.session.close()
    
    @cache.memoize(timeout=3600)
    async def fetch_page(self, url):
        """–ê—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–∞—è –∑–∞–≥—Ä—É–∑–∫–∞ —Å—Ç—Ä–∞–Ω–∏—Ü—ã"""

            async with self.session.get(url, timeout=30) as response:
                if response.status == 200:
                    return await response.text()
                return None
        except Exception as e:
            print(f"Error fetching {url}: {e}")
            return None
    
    async def analyze_domain(self, domain_id):
        """–ü–æ–ª–Ω—ã–π –∞–Ω–∞–ª–∏–∑ –¥–æ–º–µ–Ω–∞"""
        domain = Domain.query.get(domain_id)
        if not domain:
            return None
        
        # –°–æ–∑–¥–∞–Ω–∏–µ –∑–∞–ø–∏—Å–∏ –∞–Ω–∞–ª–∏–∑–∞
        analysis = Analysis(domain_id=domain_id, status='processing')
        db.session.add(analysis)
        db.session.commit()
        
        try:
            # –ó–∞–≥—Ä—É–∑–∫–∞ –≥–ª–∞–≤–Ω–æ–π —Å—Ç—Ä–∞–Ω–∏—Ü—ã
            html = await self.fetch_page(domain.url)
            if not html:
                analysis.status = 'failed'
                analysis.error = 'Failed to fetch page'
                db.session.commit()
                return None
            
            # –ü–∞—Ä—Å–∏–Ω–≥ HTML
            soup = BeautifulSoup(html, 'html.parser')
            
            # –ë–∞–∑–æ–≤—ã–π SEO –∞–Ω–∞–ª–∏–∑
            seo_data = await self._analyze_seo(soup, domain.url)
            
            # AI –∞–Ω–∞–ª–∏–∑ —Å Ollama

            # –û–±–Ω–æ–≤–ª–µ–Ω–∏–µ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤
            analysis.status = 'completed'
            analysis.results = {
                'seo_data': seo_data,
                'ai_insights': ai_insights,
                'timestamp': analysis.created_at.isoformat()
            }

            domain.seo_data = seo_data
            domain.status = 'analyzed'

            return analysis
            
        except Exception as e:
            analysis.status = 'failed'
            analysis.error = str(e)
            db.session.commit()
            return None
